    import javax.swing.*;

public class JavaFMS extends JFrame {

    //1. Data Models
	
    private static class Flight {
        String fNo;
        String name;
        int sNo;
        String firstC;
        String ecoC;
        String destination;
        String departureTime;
        String arrivalTime;

        public Flight(String fNo, String name, int sNo, String fc, String ec, String dest, String dt, String at) {
            this.fNo = fNo; this.name = name; this.sNo = sNo; this.firstC = fc; 
            this.ecoC = ec; this.destination = dest; this.departureTime = dt; this.arrivalTime = at;
        }
    }

    private static class Account {
        String username;
        String password;
        String role;
        public Account(String u, String p, String r) {
            this.username = u; this.password = p; this.role = r;
        }
    }

    private static List ACCOUNTS = new ArrayList<>();

        ACCOUNTS.add(new Account("admin", "123", "Admin")); 
        ACCOUNTS.add(new Account("user1", "pass", "User"));   
    }

    private JPanel createLoginPanel() {
        JPanel panel = new JPanel(new GridBagLayout());
        panel.setBorder(BorderFactory.createEmptyBorder(50, 50, 50, 50));
        
        JLabel title = new JLabel("Global Air FMS Login", SwingConstants.CENTER);
        title.setFont(new Font("SansSerif", Font.BOLD, 24));

        loginUserField = new JTextField(15);
        loginPassField = new JPasswordField(15);
        loginRoleCombo = new JComboBox<>(new String[]{"Admin", "User"});
        
        JButton loginButton = new JButton("Login");
        JButton signupButton = new JButton("Create Account");

        loginButton.addActionListener(new ActionListener() {
         
            public void actionPerformed(ActionEvent e) {
                attemptLogin();
            }
        });
        
        signupButton.addActionListener(new ActionListener() { 
            
            public void actionPerformed(ActionEvent e) {
                createAccountDialog();
            }
        });

        GridBagConstraints gbc = new GridBagConstraints();
        gbc.insets = new Insets(8, 8, 8, 8);
        gbc.anchor = GridBagConstraints.WEST;
        
        gbc.gridx = 0; gbc.gridy = 0; gbc.gridwidth = 2;
        panel.add(title, gbc);
        
        gbc.gridwidth = 1;
        gbc.gridy = 1; panel.add(new JLabel("Username:"), gbc);
        gbc.gridx = 1; panel.add(loginUserField, gbc);
        
        gbc.gridx = 0; gbc.gridy = 2; panel.add(new JLabel("Password:"), gbc);
        gbc.gridx = 1; panel.add(loginPassField, gbc);
        
        gbc.gridx = 0; gbc.gridy = 3; panel.add(new JLabel("Role:"), gbc);
        gbc.gridx = 1; panel.add(loginRoleCombo, gbc);
        
        gbc.gridx = 0; gbc.gridy = 4; gbc.gridwidth = 2; gbc.fill = GridBagConstraints.HORIZONTAL;
        panel.add(loginButton, gbc);
        
        gbc.gridy = 5;
        panel.add(signupButton, gbc); 

        return panel;
    }
    
    private void attemptLogin() {
        String username = loginUserField.getText().trim();
        String password = new String(loginPassField.getPassword());
        String role = (String) loginRoleCombo.getSelectedItem();
        
        for (Account acc : ACCOUNTS) {
            if (acc.username.equals(username) && acc.password.equals(password) && acc.role.equals(role)) {
                JOptionPane.showMessageDialog(this, "Login Successful! Welcome, " + username + ".", "Success", JOptionPane.INFORMATION_MESSAGE);
                mainCardLayout.show(mainPanel, role);
                loginPassField.setText("");
                return;
            }
        }
        JOptionPane.showMessageDialog(this, "Invalid username, password, or role selection.", "Login Failed", JOptionPane.ERROR_MESSAGE);
    }
    
    /*Creating a new Admin or User account. */
	
    private void createAccountDialog() {
        JTextField userField = new JTextField(15);
        JPasswordField passField = new JPasswordField(15);
        JComboBox roleCombo = new JComboBox<>(new String[]{"User", "Admin"}); 

        JPanel panel = new JPanel(new GridLayout(4, 2, 5, 5));
        panel.add(new JLabel("New Username:"));
        panel.add(userField);
        panel.add(new JLabel("New Password:"));
        panel.add(passField);
        panel.add(new JLabel("Account Role:"));
        panel.add(roleCombo);
        
        int result = JOptionPane.showConfirmDialog(this, panel, 
            "Create New Account", JOptionPane.OK_CANCEL_OPTION, JOptionPane.PLAIN_MESSAGE);

        if (result == JOptionPane.OK_OPTION) {
            String username = userField.getText().trim();
            String password = new String(passField.getPassword());
            String role = (String) roleCombo.getSelectedItem();
            
            if (username.isEmpty() || password.isEmpty()) {
                JOptionPane.showMessageDialog(this, "Username and Password cannot be empty.", "Error", JOptionPane.ERROR_MESSAGE);
                return;
            }
            
           
            for (Account acc : ACCOUNTS) {
                if (acc.username.equalsIgnoreCase(username)) {
                    JOptionPane.showMessageDialog(this, "Username '" + username + "' already exists.", "Error", JOptionPane.ERROR_MESSAGE);
                    return;
                }
            }
            
            
            ACCOUNTS.add(new Account(username, password, role));
            JOptionPane.showMessageDialog(this, role + " account created successfully for " + username + "!", "Success", JOptionPane.INFORMATION_MESSAGE);
            
        }
    }

    private void saveNewFlight() {
        try {
            if (afnField.getText().trim().isEmpty() || aNameField.getText().trim().isEmpty() || 
                !aSNoField.getText().matches("\\d+")) {
                JOptionPane.showMessageDialog(this, "Flight Code, Name, and Seat Number (Numeric) are required.", "Input Error", JOptionPane.ERROR_MESSAGE);
                return;
            }
            
            Flight newFlight = new Flight(
                afnField.getText().toUpperCase(), aNameField.getText(), 
                Integer.parseInt(aSNoField.getText()), 
                aFCField.getText(), aECField.getText(), 
                aDestField.getText(), aDTField.getText(), aATField.getText()
            );
            
            FLIGHT_RECORDS.add(newFlight);
            JOptionPane.showMessageDialog(this, "Flight " + newFlight.fNo + " added successfully.", "Success", JOptionPane.INFORMATION_MESSAGE);
            
           
            afnField.setText(""); 
            aNameField.setText(""); 
            aSNoField.setText(""); 
            aFCField.setText(""); 
            aECField.setText(""); 
            aDestField.setText(""); 
            aDTField.setText(""); 
            aATField.setText(""); 
            
        } catch (NumberFormatException ex) {
            JOptionPane.showMessageDialog(this, "Seat Number must be a valid integer.", "Input Error", JOptionPane.ERROR_MESSAGE);
        }
    }
    
    private void deleteFlight(String fNo) {
        if (fNo.trim().isEmpty()) return;
        
        boolean removed = FLIGHT_RECORDS.removeIf(f -> f.fNo.equalsIgnoreCase(fNo));
        
        if (removed) {
            JOptionPane.showMessageDialog(this, "Flight " + fNo.toUpperCase() + " deleted successfully.", "Deletion Success", JOptionPane.INFORMATION_MESSAGE);
        } else {
            JOptionPane.showMessageDialog(this, "Flight " + fNo.toUpperCase() + " not found.", "Deletion Failed", JOptionPane.ERROR_MESSAGE);
        }
    }
    
    private void searchAndDisplayFlight(String fNo) {
        if (fNo.trim().isEmpty()) return;
        
        StringBuilder sb = new StringBuilder();
        sb.append(String.format("%-10s %-15s %-10s %-10s\n", "F_NO", "NAME", "SEATS", "DESTINATION"));
        sb.append("---------------------------------------------------\n");
        
        boolean found = false;
        for (Flight f : FLIGHT_RECORDS) {
            if (f.fNo.equalsIgnoreCase(fNo)) {
                 sb.append(String.format("%-10s %-15s %-10d %-10s\n", f.fNo, f.name, f.sNo, f.destination));
                 found = true;
                 break;
            }
        }
        
        if (found) {
            JOptionPane.showMessageDialog(this, sb.toString(), "Search Result", JOptionPane.PLAIN_MESSAGE);
        } else {
             JOptionPane.showMessageDialog(this, "Flight " + fNo.toUpperCase() + " not found.", "Search Failed", JOptionPane.WARNING_MESSAGE);
        }
    }
    
    private void displayFlights() {
        StringBuilder sb = new StringBuilder();
        
        String headerLine = "=".repeat(100); 

        sb.append(headerLine).append("\n");
        sb.append(String.format("%-8s %-15s %-6s %-10s %-10s %-15s %-10s %-10s\n", 
            "F_NO", "NAME", "SEATS", "FIRST_C", "ECO_C", "DEST", "DEP_TIME", "ARR_TIME"));
        sb.append(headerLine).append("\n"); 

        if (FLIGHT_RECORDS.isEmpty()) {
            sb.append("\nNo flight records found.\n");
        } else {
            for (Flight f : FLIGHT_RECORDS) {
                sb.append(String.format("%-8s %-15s %-6d %-10s %-10s %-15s %-10s %-10s\n", 
                    f.fNo, f.name, f.sNo, f.firstC, f.ecoC, f.destination, f.departureTime, f.arrivalTime));
            }
        }
        
        displayArea.setText(sb.toString());
    }

    private void searchFlightsForUser() {
        String fNo = JOptionPane.showInputDialog(this, "Enter Flight Code or Destination:");
        if (fNo != null && !fNo.trim().isEmpty()) {
            
            StringBuilder sb = new StringBuilder();
            sb.append("SEARCH RESULTS:\n");
            
            for (Flight f : FLIGHT_RECORDS) {
                if (f.fNo.equalsIgnoreCase(fNo) || f.destination.equalsIgnoreCase(fNo)) {
                    sb.append(String.format("F_NO: %s, Route: %s -> %s, Seats: %d\n", f.fNo, f.name, f.destination, f.sNo));
                }
            }
            if (sb.length() > "SEARCH RESULTS:\n".length()) {
                JOptionPane.showMessageDialog(this, sb.toString(), "Flight Search", JOptionPane.INFORMATION_MESSAGE);
            } else {
                JOptionPane.showMessageDialog(this, "No flights found matching your query.", "Flight Search", JOptionPane.WARNING_MESSAGE);
            }
        }
    }

    private void bookFlight() {
        String fNo = ufnField.getText().toUpperCase();
        String name = uNameField.getText().trim();
        
        if (fNo.isEmpty() || name.isEmpty() || !name.matches("[a-zA-Z\\s]+")) {
            JOptionPane.showMessageDialog(this, "Flight Number and valid Passenger Name are required.", "Input Error", JOptionPane.ERROR_MESSAGE);
            return;
        }

        for (Flight f : FLIGHT_RECORDS) {
            if (f.fNo.equals(fNo)) {
                if (f.sNo > 0) {
                    f.sNo--; 
                    
                    displayBoardingPass(f, name, uSeatField.getText().toUpperCase());
                    
                    // Clear fields after booking
                    ufnField.setText(""); uNameField.setText(""); uPassportField.setText(""); uSeatField.setText("");
                    return;
                } else {
                    JOptionPane.showMessageDialog(this, "Sorry, Flight " + fNo + " is fully booked.", "Booking Failed", JOptionPane.WARNING_MESSAGE);
                    return;
                }
            }
        }
        
        JOptionPane.showMessageDialog(this, "Flight " + fNo + " not found.", "Booking Failed", JOptionPane.ERROR_MESSAGE);
    }
    
    private void displayBoardingPass(Flight f, String passengerName, String seat) {
        String pass = String.format(
            "=========================================\n" +
            "|      * Global Air Boarding Pass * |\n" +
            "=========================================\n" +
            "| FLIGHT: %-30s |\n" +
            "| ROUTE:  %-15s -> %-10s |\n" +
            "| DEPART: %-30s |\n" +
            "-----------------------------------------\n" +
            "| PASSENGER: %-25s |\n" +
            "| SEAT:      %-10s GATE: %-13s |\n" +
            "| CLASS:     %-25s |\n" +
            "=========================================\n",
            f.fNo, f.name, f.destination, f.departureTime, 
            passengerName.toUpperCase(), seat, "G14", 
            seat.contains("F") ? "FIRST" : "ECONOMY" 
        );
        
        JOptionPane.showMessageDialog(this, pass, "BOARDING PASS", JOptionPane.PLAIN_MESSAGE);
    }
